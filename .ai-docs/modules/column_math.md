# column_math

/**
 * Модуль расчета параметров ректификационной колонны на основе эталонной модели.
 * Предоставляет функцию для вычисления ключевых характеристик колонны: мощностей, потоков, теоретических тарелок.
 * Расчеты основаны на конфигурации из SamSetup и типе сырья.
 *
 * Ключевые структуры данных
 * ColumnResults — Результаты расчета параметров колонны
 */

struct ColumnResults {
  float floodPowerW;         ///< Мощность захлеба колонны, Вт
  float workingPowerW;       ///< Рабочая мощность, Вт
  float headsPowerW;         ///< Мощность для отбора голов, Вт
  float bodyEndPowerW;       ///< Мощность для окончания отбора тела, Вт
  float tailsPowerW;         ///< Мощность для отбора хвостов, Вт
  float headsFlowMlH;        ///< Поток отбора голов, мл/ч
  float bodyFlowMaxMlH;      ///< Максимальный поток отбора тела, мл/ч
  float bodyFlowMinMlH;      ///< Минимальный поток отбора тела, мл/ч
  float bodyEndFlowMlH;      ///< Поток отбора в конце тела, мл/ч
  float tailsFlowMlH;        ///< Поток отбора хвостов, мл/ч
  float maxFlowMlH;          ///< Паспортный максимальный поток (при ФЧ=4), мл/ч
  float theoreticalPlates;   ///< Количество теоретических тарелок
};
---
ColumnResults calculate_column_etalon(uint8_t rawMaterial)
Рассчитывает параметры колонны по эталонной модели на основе типа сырья.
Аргументы
rawMaterial — Тип сырья: 0=Фрукты, 1=Зерно, 2=Сахар
Возвращает
Структура ColumnResults с рассчитанными параметрами колонны
Исключения
Нет исключений
